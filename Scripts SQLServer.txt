-- Criação da tabela User
CREATE TABLE [User] (
    [Id] INT IDENTITY(1,1) PRIMARY KEY,
    [FullName] NVARCHAR(100) NOT NULL,
    [Email] NVARCHAR(100) NOT NULL,
    [Phone] NVARCHAR(20) NOT NULL,
    [PreferredChannel] NVARCHAR(50) NOT NULL,
    [CreationDate] DATETIME NOT NULL
);

-- Criação da tabela Address
CREATE TABLE [Address] (
    [Id] INT IDENTITY(1,1) PRIMARY KEY,
    [UserId] INT NOT NULL,
    [CEP] NVARCHAR(10) NOT NULL,
    [Logradouro] NVARCHAR(100) NOT NULL,
    [Complemento] NVARCHAR(100),
    [Bairro] NVARCHAR(100) NOT NULL,
    [Localidade] NVARCHAR(100) NOT NULL,
    [UF] NVARCHAR(2) NOT NULL,
    [IBGE] NVARCHAR(10) NOT NULL,
    [GIA] NVARCHAR(10) NOT NULL,
    [DDD] NVARCHAR(5) NOT NULL,
    [Siafi] NVARCHAR(10) NOT NULL,
    CONSTRAINT [FK_Address_User] FOREIGN KEY ([UserId]) REFERENCES [User] ([Id])
);

-- Criação da view UserInfo
CREATE VIEW [UserInfo] AS
SELECT
    u.Id AS UserId,
    u.FullName,
    u.Email,
    u.Phone,
    a.CEP,
    a.Logradouro,
    a.Bairro,
    a.Localidade,
    a.UF
FROM
    [User] u
    INNER JOIN [Address] a ON u.Id = a.UserId;
